2) LIGHTWEIGHT SCROLL REVEAL (no new files)

If client/src/main.tsx exists, add ONE effect at the bottom of the file (or inside the root layout file),
wrapped in a guard to avoid SSR issues. Do not alter existing renders:

if (typeof window !== 'undefined') {
  const show = () => {
    document.querySelectorAll<HTMLElement>('.reveal').forEach(el => {
      const r = el.getBoundingClientRect();
      const enter = r.top < window.innerHeight * 0.9;
      if (enter) el.classList.add('in');
    });
  };
  ['scroll','resize','load'].forEach(ev=>window.addEventListener(ev, show, {passive:true}));
  show();
}
